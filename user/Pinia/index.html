# Pinia

# 快速开始

import {
  createPinia,
  Pinia
} from 'pinia'
import persistedstate from 'pinia-plugin-persistedstate' /* 持久化插件 */

export const pinia: Pinia = createPinia()

pinia.use(persistedstate)
